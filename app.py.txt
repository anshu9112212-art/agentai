from flask import Flask, request
from twilio.twiml.messaging_response import MessagingResponse
from google import genai
import os

app = Flask(__name__)

# Bilkul naya Gemini AI Setup (2026 Latest SDK)
client = genai.Client(api_key=os.environ.get("GEMINI_API_KEY"))

@app.route("/whatsapp", methods=['POST'])
def whatsapp_reply():
    # WhatsApp se aaya hua message read karna
    incoming_msg = request.values.get('Body', '').strip()
    
    try:
        # Naye SDK ke mutabik content generate karna
        response = client.models.generate_content(
            model='gemini-2.5-flash',
            contents=incoming_msg,
        )
        bot_reply = response.text
    except Exception as e:
        bot_reply = "Sorry, Gemini AI abhi connect nahi ho pa raha hai."

    # Jawab ko sahi tarike se Twilio format me badalna
    twilio_resp = MessagingResponse()
    twilio_resp.message(bot_reply)

    return str(twilio_resp)

if __name__ == "__main__":
    # Render port 10000 use karta hai, isliye isse auto-detect hone dete hain
    port = int(os.environ.get("PORT", 10000))
    app.run(host="0.0.0.0", port=port)
